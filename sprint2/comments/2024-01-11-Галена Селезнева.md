TABLE WITHOUT ID Total as "Общее время внесенное за день"
WHERE file.path = this.file.path 
FLATTEN file.lists as Lists
WHERE Lists.spent
GROUP BY ""
FLATTEN sum(rows.Lists.spent) as Total
